class AlexNet(nn.Module):
    def __init__(self):
        # з»§жүҝ
        super(AlexNet,self).__init__()
        self.relu = nn.ReLU() # е®ҡд№үжҝҖжҙ»еҮҪж•°
        self.c1 = nn.Conv2d(in_channels=1,out_channels=96,kernel_size=11,stride=4)
        self.s2 = nn.MaxPool2d(kernel_size=3,stride=2)
        self.c3 = nn.Conv2d(in_channels=96,out_channels=256,kernel_size=5,padding=2)
        self.s4 = nn.MaxPool2d(kernel_size=3,stride=2)
        self.c5 = nn.Conv2d(in_channels=256,out_channels=384,kernel_size=3,padding=1)
        self.c6 = nn.Conv2d(in_channels=384, out_channels=384, kernel_size=3, padding=1)
        self.c7 = nn.Conv2d(in_channels=384, out_channels=256, kernel_size=3, padding=1)
        self.s8 = nn.MaxPool2d(kernel_size=3,stride=2)
        self.flatten = nn.Flatten()
        # е…ЁиҝһжҺҘеұӮ
        self.f1 = nn.Linear(in_features=6*6*256,out_features=4096)
        self.f2 = nn.Linear(in_features=4096,out_features=4096)
        self.f3 = nn.Linear(in_features=4096,out_features=10) # жңҖеҗҺзҡ„иҫ“еҮәдёәд»»еҠЎиҖҢе®ҡ

    def forward(self,x):
        # жҜҸж¬Ўз»ҸиҝҮеҚ·з§ҜйғҪеә”иҜҘжңүдёӘreluеҮҪж•°жҝҖжҙ»
        x = self.relu(self.c1(x))
        x = self.s2(x)
        x = self.relu(self.c3(x))
        x = self.s4(x)
        x = self.relu(self.c5(x))
        x = self.relu(self.c6(x))
        x = self.relu(self.c7(x))
        x = self.s8(x)

class AlexNet(nn.Module):
    def __init__(self):
        # з»§жүҝ
        super(AlexNet,self).__init__()
        self.relu = nn.ReLU() # е®ҡд№үжҝҖжҙ»еҮҪж•°
        self.c1 = nn.Conv2d(in_channels=1,out_channels=96,kernel_size=11,stride=4)
        self.s2 = nn.MaxPool2d(kernel_size=3,stride=2)
        self.c3 = nn.Conv2d(in_channels=96,out_channels=256,kernel_size=5,padding=2)
        self.s4 = nn.MaxPool2d(kernel_size=3,stride=2)
        self.c5 = nn.Conv2d(in_channels=256,out_channels=384,kernel_size=3,padding=1)
        self.c6 = nn.Conv2d(in_channels=384, out_channels=384, kernel_size=3, padding=1)
        self.c7 = nn.Conv2d(in_channels=384, out_channels=256, kernel_size=3, padding=1)
        self.s8 = nn.MaxPool2d(kernel_size=3,stride=2)
        self.flatten = nn.Flatten()
        # е…ЁиҝһжҺҘеұӮ
        self.f1 = nn.Linear(in_features=6*6*256,out_features=4096)
        self.f2 = nn.Linear(in_features=4096,out_features=4096)
        self.f3 = nn.Linear(in_features=4096,out_features=10) # жңҖеҗҺзҡ„иҫ“еҮәдёәд»»еҠЎиҖҢе®ҡ

    def forward(self,x):

        x = self.relu(self.c1(x))
        x = self.s2(x)
        x = self.relu(self.c3(x))
        x = self.s4(x)
        x = self.relu(self.c5(x))
        x = self.relu(self.c6(x))
        x = self.relu(self.c7(x))
        x = self.s8(x)

        x = self.flatten(x)

        x = self.relu(self.f1(x))
        x = F.dropout(x,0.5) # зҘһз»Ҹе…ғйҡҸжңәеӨұжҙ»пјҢжңү0.5зҡ„зҘһз»Ҹе…ғеӨұжҙ»

        x = F.dropout(x,0.5)
        x = self.f3(x)
        return x

class VGG16(nn.Module):
    def __init__(self):
        super(VGG16,self).__init__()

        self.block1 = nn.Sequential(
            nn.Conv2d(in_channels=1,out_channels=64,kernel_size=3,padding=1,stride=1),
            nn.ReLU(),
            nn.Conv2d(in_channels=64, out_channels=64, kernel_size=3, padding=1, stride=1),
            nn.ReLU(),
            nn.MaxPool2d(kernel_size=2, stride=2),
        )
        self.block2 = nn.Sequential(
            nn.Conv2d(in_channels=64,out_channels=128,kernel_size=3,padding=1,stride=1),
            nn.ReLU(),
            nn.Conv2d(in_channels=128, out_channels=128, kernel_size=3, padding=1, stride=1),
            nn.ReLU(),
            nn.MaxPool2d(kernel_size=2, stride=2),
        )
        self.block3 = nn.Sequential(

            nn.ReLU(),
            nn.Conv2d(in_channels=256, out_channels=256, kernel_size=3, padding=1, stride=1),
            nn.ReLU(),
            nn.Conv2d(in_channels=256, out_channels=256, kernel_size=3, padding=1, stride=1),
            nn.ReLU(),
            nn.MaxPool2d(kernel_size=2, stride=2),
        )
        self.block4 = nn.Sequential(
            nn.Conv2d(in_channels=256,out_channels=512,kernel_size=3,padding=1,stride=1),
            nn.ReLU(),
            nn.Conv2d(in_channels=512, out_channels=512, kernel_size=3, padding=1, stride=1),
            nn.ReLU(),
            nn.Conv2d(in_channels=512, out_channels=512, kernel_size=3, padding=1, stride=1),
            nn.ReLU(),
            nn.MaxPool2d(kernel_size=2, stride=2),
        )
        self.block5 = nn.Sequential(
            nn.Conv2d(in_channels=512,out_channels=512,kernel_size=3,padding=1,stride=1),
            nn.ReLU(),
            nn.Conv2d(in_channels=512, out_channels=512, kernel_size=3, padding=1, stride=1),
            nn.ReLU(),
            nn.Conv2d(in_channels=512, out_channels=512, kernel_size=3, padding=1, stride=1),
            nn.ReLU(),
            nn.MaxPool2d(kernel_size=2, stride=2),
        )
        # е…ЁиҝһжҺҘеұӮ
        self.block6 = nn.Sequential(

            nn.ReLU(),

            nn.ReLU(),
            nn.Linear(in_features=4096,out_features=10),
        )

        # еҸӮж•°еҲқе§ӢеҢ–
        for m in self.modules():
            if isinstance(m,nn.Conv2d):
                nn.init.kaiming_normal_(m.weight,nonlinearity='relu')
                if m.bias is not None:
                    nn.init.constant_(m.bias,0)
            elif isinstance(m, nn.Linear):
                nn.init.normal_(m.weight,0,0.01)
                if m.bias is not None:
                    nn.init.constant_(m.bias,0)

        x = self.block2(x)
        x = self.block3(x)
        x = self.block4(x)
        x = self.block5(x)
        x = self.block6(x)

        return x

class Inception(nn.Module):
    def __init__(self,in_channels,c1,c2,c3,c4): # c1,c2,c3,c4дёәиҫ“еҮәйҖҡйҒ“ж•°е’Ңиҫ“еҮәйҖҡйҒ“ж•°зҡ„ж•°йҮҸ
        super(Inception,self).__init__()
        self.rulu = nn.ReLU()

        # и·Ҝзәҝ1,еҚ•1*1еҚ·з§ҜеұӮ
        self.p1 = nn.Conv2d(in_channels=in_channels,out_channels=c1,kernel_size=1)

        # и·Ҝзәҝ2, 1*1еҚ·з§ҜпјҢ3*3еҚ·з§Ҝ
        self.p2_1 = nn.Conv2d(in_channels=in_channels,out_channels=c2[0],kernel_size=1)
        self.p2_2 = nn.Conv2d(in_channels=c2[0],out_channels=c2[1],kernel_size=3,padding=1)

        # и·Ҝзәҝ3, 1*1еҚ·з§ҜпјҢ5*5еҚ·з§Ҝ
        self.p3_1 = nn.Conv2d(in_channels=in_channels,out_channels=c3[0],kernel_size=1)
        self.p3_2 = nn.Conv2d(in_channels=c3[0],out_channels=c3[1],kernel_size=5,padding=2)

        # и·Ҝеҫ„4,  3*3жңҖеӨ§жұ еҢ–,1*1еҚ·з§Ҝ
        self.p4_1 = nn.MaxPool2d(kernel_size=3,padding=1,stride=1) # жұ еҢ–зҡ„жӯҘе№…дјҡе’Ңkernel_sizeзӣёзӯү
        self.p4_2 = nn.Conv2d(in_channels=in_channels,out_channels=c4,kernel_size=1)

    # еӣ дёәжҳҜе№¶иЎҢзҡ„з»“жһ„
    def forward(self,x):
        p1 = self.rulu(self.p1(x))
        # и·Ҝеҫ„2
        p2 = self.rulu(self.p2_2(self.rulu(self.p2_1(x))))
        # и·Ҝзәҝ3
        p3 = self.rulu(self.p3_2(self.rulu(self.p3_1(x))))
        # и·Ҝзәҝ4
        p4 = self.rulu(self.p4_2(self.p4_1(x)))

        return torch.cat((p1,p2,p3,p4),dim=1) # зү№еҫҒиһҚеҗҲ,dim=1жҢҮе®ҡдёәйҖҡйҒ“

stride  жӯҘе№…
padding еЎ«е……

'''

class AlexNet(nn.Module):
    def __init__(self):
        # з»§жүҝ
        super(AlexNet,self).__init__()
        self.relu = nn.ReLU() # е®ҡд№үжҝҖжҙ»еҮҪж•°
        self.c1 = nn.Conv2d(in_channels=1,out_channels=96,kernel_size=11,stride=4)
        self.s2 = nn.MaxPool2d(kernel_size=3,stride=2)
        self.c3 = nn.Conv2d(in_channels=96,out_channels=256,kernel_size=5,padding=2)
        self.s4 = nn.MaxPool2d(kernel_size=3,stride=2)
        self.c5 = nn.Conv2d(in_channels=256,out_channels=384,kernel_size=3,padding=1)
        self.c6 = nn.Conv2d(in_channels=384, out_channels=384, kernel_size=3, padding=1)
        self.c7 = nn.Conv2d(in_channels=384, out_channels=256, kernel_size=3, padding=1)
        self.s8 = nn.MaxPool2d(kernel_size=3,stride=2)
        self.flatten = nn.Flatten()
        # е…ЁиҝһжҺҘеұӮ
        self.f1 = nn.Linear(in_features=6*6*256,out_features=4096)
        self.f2 = nn.Linear(in_features=4096,out_features=4096)
        self.f3 = nn.Linear(in_features=4096,out_features=10) # жңҖеҗҺзҡ„иҫ“еҮәдёәд»»еҠЎиҖҢе®ҡ

    def forward(self,x):
        # жҜҸж¬Ўз»ҸиҝҮеҚ·з§ҜйғҪеә”иҜҘжңүдёӘreluеҮҪж•°жҝҖжҙ»
        x = self.relu(self.c1(x))
        x = self.s2(x)
        x = self.relu(self.c3(x))
        x = self.s4(x)
        x = self.relu(self.c5(x))
        x = self.relu(self.c6(x))
        x = self.relu(self.c7(x))
        x = self.s8(x)

        x = self.flatten(x)

        x = self.relu(self.f1(x))
        x = F.dropout(x,0.5) # зҘһз»Ҹе…ғйҡҸжңәеӨұжҙ»пјҢжңү0.5зҡ„зҘһз»Ҹе…ғеӨұжҙ»
        x = self.relu(self.f2(x))
        x = F.dropout(x,0.5)
        x = self.f3(x)
        return x

if __name__ == '__main__':
    device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
